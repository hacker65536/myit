
https://raw.githubusercontent.com/kubernetes/kubernetes/v1.10.3/examples/guestbook-go/redis-master-controller.json
```json
{
   "kind":"ReplicationController",
   "apiVersion":"v1",
   "metadata":{
      "name":"redis-master",
      "labels":{
         "app":"redis",
         "role":"master"
      }
   },
   "spec":{
      "replicas":1,
      "selector":{
         "app":"redis",
         "role":"master"
      },
      "template":{
         "metadata":{
            "labels":{
               "app":"redis",
               "role":"master"
            }
         },
         "spec":{
            "containers":[
               {
                  "name":"redis-master",
                  "image":"redis:2.8.23",
                  "ports":[
                     {
                        "name":"redis-server",
                        "containerPort":6379
                     }
                  ]
               }
            ]
         }
      }
   }
}
```
